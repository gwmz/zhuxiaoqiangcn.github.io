### 汉诺塔

```C
#include <stdio.h>
#include <stdlib.h>
void move(int n,char a,char b, char c)
{
    if(n==1)//1个环的情况，也是递归的出口
    {
        printf("%c-->%c\n",a,c);
    }
    else//n > 1时，用抽象出的3步来移动
    {
        move(n-1,a,c,b);//将n-1个环从a移动到b上
        printf("%c-->%c\n",a,c);//将底盘从a移动到c上
        move(n-1,b,a,c);//将b上的n-1个环移动到c上
    }
}

int main()
{
    move(4,'A','B','C');
    return 0;
}
```
